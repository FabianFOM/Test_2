let selectedFolder = "";

// Skripte laden
async function loadScripts() {
  const res = await fetch("http://localhost:3000/scripts");
  const data = await res.json();

  const list = document.getElementById("scripts");
  list.innerHTML = "";

  data.forEach(s => {
    const li = document.createElement("li");
    li.innerText = s;
    li.onclick = () => {
      selectedFolder = s;
      alert("Gewählt: " + s);
    };
    list.appendChild(li);
  });
}

// Upload
async function upload() {
  const file = document.getElementById("file").files[0];
  const formData = new FormData();
  formData.append("pdf", file);

  await fetch("http://localhost:3000/upload", {
    method: "POST",
    body: formData,
  });

  loadScripts();
}

// Frage
async function ask() {
  const question = document.getElementById("question").value;

  const res = await fetch("http://localhost:3000/ask", {
    method: "POST",
    headers: {"Content-Type": "application/json"},
    body: JSON.stringify({
      folder: selectedFolder,
      question
    }),
  });

  const data = await res.json();
  document.getElementById("answer").innerText = data.answer;
}

// Quiz
async function generateQuiz() {
  const res = await fetch("http://localhost:3000/test", {
    method: "POST",
    headers: {"Content-Type": "application/json"},
    body: JSON.stringify({ folder: selectedFolder }),
  });

  const data = await res.json();

  let quizData;
  try {
    quizData = JSON.parse(data.test);
  } catch {
    alert("Fehler beim Quiz – nochmal versuchen");
    return;
  }

  const quizDiv = document.getElementById("quiz");
  quizDiv.innerHTML = "";

  quizData.forEach(q => {
    const qDiv = document.createElement("div");
    qDiv.innerHTML = "" + q.frage + "";

    q.optionen.forEach(opt => {
      const btn = document.createElement("div");
      btn.innerText = opt;
      btn.className = "quiz-option";

      btn.onclick = () => {
        if (opt === q.korrekt) {
          btn.classList.add("correct");
        } else {
          btn.classList.add("wrong");
        }
      };

      qDiv.appendChild(btn);
    });

    quizDiv.appendChild(qDiv);
  });
}
